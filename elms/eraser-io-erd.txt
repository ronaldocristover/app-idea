// relationships

// Users ↔ Roles (many-to-many via user_roles)
user_roles.user_id > users.user_id
user_roles.role_id > roles.role_id

// Roles ↔ Permissions (many-to-many via role_permissions)
role_permissions.role_id > roles.role_id
role_permissions.permission_id > permissions.permission_id

// Users ↔ Courses (many-to-many via Enrollments)
enrollments.user_id > users.user_id
enrollments.course_id > courses.course_id

// Courses ↔ Modules (one-to-many)
modules.course_id > courses.course_id

// Courses ↔ Course Categories (one-to-many)
courses.category_id > course_categories.category_id

// Modules ↔ Lessons (one-to-many)
lessons.module_id > modules.module_id

// Lessons ↔ Quizzes (one-to-many)
quizzes.lesson_id > lessons.lesson_id

// Lessons ↔ Assignments (one-to-many)
assignments.lesson_id > lessons.lesson_id

// Quizzes ↔ Questions (one-to-many)
questions.quiz_id > quizzes.quiz_id

// Questions ↔ Answers (one-to-many)
answers.question_id > questions.question_id

// Users ↔ Submissions (many-to-many)
submissions.user_id > users.user_id
submissions.quiz_id > quizzes.quiz_id

// Assignments ↔ assignment_submissions (one-to-many)
assignment_submissions.assignment_id > assignments.assignment_id
assignment_submissions.user_id > users.user_id

// Users ↔ Badges (many-to-many via user_badges)
user_badges.user_id > users.user_id
user_badges.badge_id > badges.badge_id

// Orders ↔ Courses (many-to-many via order_items)
order_items.order_id > orders.order_id
order_items.course_id > courses.course_id

// Orders ↔ Payments (one-to-many)
payments.order_id > orders.order_id

// Users ↔ Subscriptions ↔ Plans (one-to-many)
subscriptions.user_id > users.user_id
subscriptions.plan_id > plans.plan_id

// Courses ↔ Instructors (many-to-many)
revenue_sharing.course_id > courses.course_id
revenue_sharing.instructor_id > instructors.instructor_id
instructors.user_id > users.user_id

// Instructors ↔ revenue_sharing (one-to-many)
revenue_sharing.instructor_id > instructors.instructor_id

// Instructors ↔ Payouts (one-to-many)
payouts.instructor_id > instructors.instructor_id

// Users ↔ Certificates (one-to-many)
certificates.user_id > users.user_id
certificates.course_id > courses.course_id

// course_categories (self-referencing)
course_categories.parent_id > course_categories.category_id

// Announcements ↔ Courses
announcements.course_id > courses.course_id

// Announcements ↔ Users (created_by)
announcements.created_by > users.user_id

// Forums ↔ Courses
forums.course_id > courses.course_id

// Forums ↔ Users (created_by)
forums.created_by > users.user_id

// Posts ↔ Forums
posts.forum_id > forums.forum_id

// Posts ↔ Users
posts.user_id > users.user_id

// Comments ↔ Posts
comments.post_id > posts.post_id

// Comments ↔ Users
comments.user_id > users.user_id

// Gradebook ↔ Users & Courses
gradebook.user_id > users.user_id
gradebook.course_id > courses.course_id

// auth_providers ↔ Users
auth_providers.user_id > users.user_id

// user_security ↔ Users
user_security.user_id > users.user_id

// assignment_submissions ↔ Assignments
assignment_submissions.assignment_id > assignments.assignment_id

// assignment_submissions ↔ Users
assignment_submissions.user_id > users.user_id

// Points ↔ Users
points.user_id > users.user_id

// Messages ↔ Users (sender/receiver)
messages.sender_id > users.user_id
messages.receiver_id > users.user_id

// user_activity_logs ↔ Users
user_activity_logs.user_id > users.user_id

// Orders ↔ Users
orders.user_id > users.user_id

// order_items ↔ Orders
order_items.order_id > orders.order_id

// order_items ↔ Courses
order_items.course_id > courses.course_id

// Payments ↔ Orders
payments.order_id > orders.order_id

// Submissions ↔ Quizzes
submissions.quiz_id > quizzes.quiz_id

// Submissions ↔ Users
submissions.user_id > users.user_id

// assignment_submissions ↔ Assignments
assignment_submissions.assignment_id > assignments.assignment_id

// assignment_submissions ↔ Users
assignment_submissions.user_id > users.user_id

// Certificates ↔ Users & Courses
certificates.user_id > users.user_id
certificates.course_id > courses.course_id

// revenue_sharing ↔ Courses & Instructors
revenue_sharing.course_id > courses.course_id
revenue_sharing.instructor_id > instructors.instructor_id

// Payouts ↔ Instructors
payouts.instructor_id > instructors.instructor_id
users.default_language_id > languages.language_id
course_localizations.course_id > courses.course_id
course_localizations.language_id > languages.language_id
post_localizations.post_id > posts.post_id
post_localizations.language_id > languages.language_id
